<script>
if(document.readyState != 'loading')
  initmonthsQuiz();
else
  document.addEventListener("DOMContentLoaded", initmonthsQuiz);

function initmonthsQuiz() {
  const quizEl = document.getElementById("months-quiz");
  if (!quizEl) return; // not on this page
  window.monthsQuiz = new MatchingQuiz(
    "months-quiz",
    [
      { left: "1st month", right: "လါသလ့ၤ" },
      { left: "8th month", right: "လါခူး" },
      { left: "12th month", right: "လါပျုၤ" },
      { left: "Rainy season", right: "တၢ်စူၤခါ" }
    ],
    () =>
      document
        .getElementById("post-quiz")
        ?.classList.remove("display-none")
  );

  window.checkmonthsQuiz = function () {
    const score = monthsQuiz.check();
    const result = document.getElementById("months-result");

    result.textContent = `You got ${score} / 4 correct.`;
    result.style.color = score === 4 ? "green" : "red";
  };
}

</script>
