document.addEventListener("DOMContentLoaded", () => {

  window.daysQuiz = new MatchingQuiz(
    "days-quiz",
    [
      { left: "Friday", right: "မုၢ်ယဲၢ်နံၤ" },
      { left: "Tomorrow", right: "ခဲမုၢ်ဆ့ၣ်" },
      { left: "Yesterday", right: "မဟါကၢာ် / မဟါတနံၤ" },

    ],
    () => document.getElementById("post-quiz")
      .classList.remove("display-none")
  );

  window.checkDaysQuiz = function () {
    const score = daysQuiz.check();
    const result = document.getElementById("days-result");

    result.textContent = `You got ${score} / 4 correct.`;
    result.style.color = score === 4 ? "green" : "red";
  };

});
