<h3>Example breakdown</h3>
The video used the example:

ယမံၤမ့ၢ်နီၢ်ဖီ meaning "My name is Paw"

Syllable by syllable this breaks down as ယ (my) မံၤ (name) မ့ၢ် (is) နီၢ် (female name marker) ဖီ (flower / paw)

if(document.readyState != 'loading')

function initinterrogativesQuiz() {
  const quizEl = document.getElementById("interrogatives-quiz");
  if (!quizEl) return; // not on this page
  window.interrogativesQuiz = new MatchingQuiz(
    "interrogatives-quiz",
    [
      { left: "How?", right: "ဒ်လဲၣ်" },
      { left: "Why?", right: "ဘၣ်မနုၤလဲၣ်" },
      { left: "Who?", right: "မတၤလဲၣ်" },
      { left: "Yes or no question marker", right: "ဧါ" }
    ],
    () =>
      document
        .getElementById("post-quiz")
        ?.classList.remove("display-none")
  );

  window.checkinterrogativesQuiz = function () {
    const score = interrogativesQuiz.check();
    const result = document.getElementById("interrogatives-result");

    result.textContent = `You got ${score} / 4 correct.`;
    result.style.color = score === 4 ? "green" : "red";
  };
}

</script>
