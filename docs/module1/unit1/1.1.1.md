<h3>Explanations / vocabulary</h3>
While memorizing phrases for greetings can work, it is useful to learn the vocabulary used within these phrases.

First, in the "good morning" and "good afternoon" type of greetings, you likely noticed they follow a consistent pattern. First, one unique syllable (which denotes the time of day) followed by "လၢအဂ့ၤ" which means "that is good" or "which is good". Here, "ဂ့ၤ" means good. The first syllables are as follows:

if(document.readyState != 'loading')

function initGreetingsQuiz() {
  const quizEl = document.getElementById("greetings-quiz");
  if (!quizEl) return; // not on this page
  window.greetingsQuiz = new MatchingQuiz(
    "greetings-quiz",
    [
      { left: "Good evening", right: "ဟါလၢအဂ့ၤ" },
      { left: "How are you?", right: "နအိၣ်ဆူၣ်အိၣ်ချ့ဧါ" },
      { left: "Have you eaten already?", right: "နအီၣ်မ့ၤဝံၤလံဧါ" },
      { left: "How's it going?", right: "တၢ်အိၣ်သးမၤသးဒ်လဲၣ်" }
    ],
    () =>
      document
        .getElementById("post-quiz")
        ?.classList.remove("display-none")
  );

  window.checkGreetingsQuiz = function () {
    const score = greetingsQuiz.check();
    const result = document.getElementById("greetings-result");

    result.textContent = `You got ${score} / 4 correct.`;
    result.style.color = score === 4 ? "green" : "red";
  };
}

</script>
