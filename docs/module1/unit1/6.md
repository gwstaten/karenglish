<script>
if(document.readyState != 'loading')
  initnumbersQuiz();
else
  document.addEventListener("DOMContentLoaded", initnumbersQuiz);

function initnumbersQuiz() {
  const quizEl = document.getElementById("numbers-quiz");
  if (!quizEl) return; // not on this page
  window.numbersQuiz = new MatchingQuiz(
    "numbers-quiz",
    [
      { left: "6", right: "ဃု" },
      { left: "2", right: "ခံ" },
      { left: "100s place holder (not money)", right: "ကယၤ" },
      { left: "100s place holder (money)", right: "ပီၢ်" }
    ],
    () =>
      document
        .getElementById("post-quiz")
        ?.classList.remove("display-none")
  );

  window.checknumbersQuiz = function () {
    const score = numbersQuiz.check();
    const result = document.getElementById("numbers-result");

    result.textContent = `You got ${score} / 4 correct.`;
    result.style.color = score === 4 ? "green" : "red";
  };
}

</script>
